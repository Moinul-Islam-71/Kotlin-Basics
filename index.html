Kotlin Cheat Sheet

1. Basic Syntax

fun main() {
    println("Hello, World!")  // Prints a message
}

Variable Declaration:

val name = "Taiyo"  // Immutable (can't change)
var age = 25        // Mutable (can change)
age = 26            // Valid for `var`

Data Types:

val num: Int = 10                 // Integer type
val pi: Double = 3.14             // Double type
val isKotlinFun: Boolean = true   // Boolean type
val letter: Char = 'K'            // Character type
val name: String = "Kotlin"       // String type

2. Control Flow

If-Else:

val a = 10
val b = 20
val max = if (a > b) a else b  // Finds the maximum of two numbers
println("Maximum: $max")  // Output the maximum value

When:

val x = 3
when (x) {
    1 -> println("One")  // Prints if x is 1
    2 -> println("Two")  // Prints if x is 2
    in 3..5 -> println("Three to Five")  // Range check for 3 to 5
    else -> println("Other")  // Prints if x is not 1, 2, or between 3 and 5
}

Loops:

For Loop:

for (i in 1..5) {  // Loops from 1 to 5
    println("Number: $i")  // Prints each number
}

for (i in 5 downTo 1 step 2) {  // Loops from 5 to 1 with step 2
    println("Step: $i")  // Prints 5, 3, 1
}

While Loop:

var i = 5
while (i > 0) {  // Loops as long as i is greater than 0
    println(i--)  // Prints i and then decrements it
}

3. Functions

Regular Function:

fun multiply(a: Int, b: Int): Int {  // Function to multiply two integers
    return a * b  // Returns the product of a and b
}

Single-Expression Function:

fun greet(name: String) = "Hello, $name!"  // Returns a greeting message
println(greet("Taiyo"))  // Prints the greeting for Taiyo

Default Arguments:

fun printInfo(name: String = "Anonymous", age: Int = 0) {  
    println("Name: $name, Age: $age")  // Prints name and age
}
printInfo("Taiyo", 25)  // Prints the provided values
printInfo()  // Uses default values for name and age

Named Arguments:

printInfo(age = 30, name = "Kotlin")  // Passes values explicitly

4. Classes and Objects

Primary Constructor with Properties:

class Person(val name: String, var age: Int) {  // Class with a primary constructor
    fun introduce() {  // Method to introduce the person
        println("Hi, I'm $name, and I'm $age years old.")  // Prints introduction
    }
}

Secondary Constructor:

class Student(val name: String) {  // Class with a secondary constructor
    var grade: String = "Unknown"  // Default grade is "Unknown"

    constructor(name: String, grade: String) : this(name) {  // Secondary constructor
        this.grade = grade  // Sets grade value
    }
}

Object Creation:

fun main() {
    val person = Person("Taiyo", 25)  // Creates a Person object
    person.introduce()  // Calls the introduce method

    val student = Student("Saitama", "A")  // Creates a Student object with grade
    println("${student.name}'s grade is ${student.grade}")  // Prints student's grade
}

5. Collections

Lists:

val immutableList = listOf("Apple", "Banana", "Cherry")  // Immutable list
val mutableList = mutableListOf("Dog", "Cat")  // Mutable list

mutableList.add("Parrot")  // Adds a new element to the mutable list
println(mutableList)  // Prints the updated list

Maps:

val immutableMap = mapOf("name" to "Taiyo", "age" to 25)  // Immutable map with key-value pairs
val mutableMap = mutableMapOf("language" to "Kotlin")  // Mutable map

mutableMap["level"] = "Advanced"  // Adds/updates a key-value pair in the mutable map
println(mutableMap)  // Prints the updated map

Sets:

val set = setOf(1, 2, 3, 3)  // Set (duplicates will be removed)
println(set)  // Prints the set (output will be {1, 2, 3})

6. Null Safety

var name: String? = null  // Nullable variable (can hold null)
println(name?.length ?: "Name is null")  // Elvis operator: if name is null, print "Name is null"

7. Extension Functions

fun String.addExclamation() = this + "!"  // Adds "!" at the end of a string
println("Kotlin".addExclamation())  // Prints "Kotlin!"

8. Higher-Order Functions

fun operate(x: Int, y: Int, operation: (Int, Int) -> Int): Int {  // Higher-order function
    return operation(x, y)  // Applies the operation function to x and y
}
val result = operate(5, 3) { a, b -> a + b }  // Passes a lambda function to add x and y
println(result)  // Prints the result of addition (8)

9. Coroutines

import kotlinx.coroutines.*

fun main() = runBlocking {
    launch {  // Launches a coroutine
        delay(1000L)  // Simulates a delay
        println("World!")  // Prints "World!"
    }
    println("Hello,")  // Prints "Hello,"
}
fun main() = runBlocking {
    val job1 = async { task1() }  // Launches async task1
    val job2 = async { task2() }  // Launches async task2

    println("Result: ${job1.await() + job2.await()}")  // Waits for both tasks to finish and prints the sum
}

suspend fun task1(): Int {
    delay(1000L)  // Simulates a delay
    return 10  // Returns 10
}

suspend fun task2(): Int {
    delay(1000L)  // Simulates a delay
    return 20  // Returns 20
}

10. Data Classes

data class User(val name: String, val age: Int)  // Data class with two properties

val user = User("Taiyo", 25)  // Creates a User object
println(user)  // Prints the user object (User(name=Taiyo, age=25))

11. Lambda Expressions


// Defining a lambda expression that squares the input
val square = { x: Int -> x * x }  

// Calling the lambda expression with 4 as the argument and printing the result
println(square(4))  // Output: 16
